        if ("Java+1" prefix langName)
            if(langName prefix "Java+1")
                "Static+1"
            else
                if(reverse "script+1" prefix reverse langName)
                    "Dynamic+1"
                else

        else
            if (reverse "script+1" prefix reverse langName)
                "Probably Dynamic+1"
            else

    }
